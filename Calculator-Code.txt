    <script>
        let currentInput = '';
        let lastOperation = '';
        let lastValue = 0;

        function appendNumber(num) {
            currentInput += num;
            document.getElementById('screen').innerText = currentInput;
        }

        function appendDecimal() {
            if (!currentInput.includes('.')) {
                currentInput += '.';
                document.getElementById('screen').innerText = currentInput;
            }
        }

        function setOperation(op) {
            if (currentInput === '') return;
            if (lastValue !== 0) {
                calculate();
            }
            lastOperation = op;
            lastValue = parseFloat(currentInput);
            currentInput = '';
        }

        function calculate() {
            let result;
            const currentNumber = parseFloat(currentInput);
            if (isNaN(currentNumber) || lastOperation === '') return;

            switch (lastOperation) {
                case '+':
                    result = lastValue + currentNumber;
                    break;
                case '-':
                    result = lastValue - currentNumber;
                    break;
                case '*':
                    result = lastValue * currentNumber;
                    break;
                case '/':
                    if (currentNumber === 0) {
                        alert("ŸÑÿß ŸäŸÖŸÉŸÜ ÿßŸÑŸÇÿ≥ŸÖÿ© ÿπŸÑŸâ ÿµŸÅÿ±!");
                        clearAll();
                        return;
                    }
                    result = lastValue / currentNumber;
                    break;
            }
            currentInput = result.toString();
            lastOperation = '';
            lastValue = 0;
            document.getElementById('screen').innerText = currentInput;
        }

        function clearAll() {
            currentInput = '';
            lastOperation = '';
            lastValue = 0;
            document.getElementById('screen').innerText = '0';
        }

        document.addEventListener('keydown', function(event) {
            const key = event.key;
            if (!isNaN(key)) {
                appendNumber(key);
            } else if (key === '.') {
                appendDecimal();
            } else if (key === '+') {
                setOperation('+');
            } else if (key === '-') {
                setOperation('-');
            } else if (key === '*') {
                setOperation('*');
            } else if (key === '/') {
                setOperation('/');
            } else if (key === 'Enter' || key === '=') {
                calculate();
            } else if (key === 'Escape') {
                clearAll();
            }
        });
